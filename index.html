Componitor

Simple directive generation for web component -like templates to encourage reuse in AngularJS

Download it here (v. {{site.componitor_version}})

Reuse

Easily create reusable directives for, e.g., layouts and ui components not needing complex logic

Reduce

Reduce repetition of html fragments to encourage the DRY principle. Easily add WAI-ARIA attributes to common components without having to type them every time.

Lazy-load

Create layout directives just-in-time by calling Componitor.process() on a piece of html

Usage

<div ng-controller="ProgressCtrl as ctrl">
  <bs-progress values="ctrl.progress"></bs-progress>
  <button class="btn btn-primary" ng-click="ctrl.randomize()">
    Randomize values
  </button>
</div>

Template

<componitor-template name="bsProgress" isolate="true" >
  <div class="progress">
    <bs-bar values="{type: 'success', value: values.success}">
    </bs-bar>
    <bs-bar values="{type: 'warning', value: values.warning}">
    </bs-bar>
  </div>
</componitor-template>
<componitor-template name="bsBar" isolate="true" >
  <div class="progress-bar progress-bar-{{values.type}}"
      style="width: {{values.value}}%"
      aria-valuenow="{{values.value}}"
      aria-valuemin="0" aria-valuemax="100">
    <span>{{values.value}}%</span>
  </div>
</componitor-template>

JavaScript

angular.module('examples', ['componitor'])
  .controller('ProgressCtrl', function() {
    this.progress = {
      success: 40,
      warning: 20
    };
    function randomInt(maxVal) {
      return Math.round(Math.random()*maxVal);
    }
    this.randomize = function() {
      this.progress.success = randomInt(100);
      this.progress.warning = randomInt(100-this.progress.success);
    };
  });
